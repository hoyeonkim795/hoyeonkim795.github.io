def calculate(start, end, n, adj):
    INF = float('inf')
    dist = [INF] * (n+1)
    selected = [False] * (n+1)

    dist[start] = 0
    cnt = 1 # ì •ì  ì‹œì‘ ì§€ì  ë¬¸ì œëŠ” 1ë¶€í„° ì‹œì‘ì´ë‹ˆê¹Œ

    while cnt < n+1:
        # dist ê°€ ìµœì†Œì¸ ì •ì  ì°¾ê¸°
        min = INF
        u = 1 # ì •ì  ì‹œì‘ ì§€ì  ë¬¸ì œëŠ” 1ë¶€í„° ì‹œì‘ì´ë‹ˆê¹Œ
        for i in range(1,n+1):
            if not selected[i] and dist[i] < min:
                min = dist[i]
                u = i
        # ê²°ì •
        selected[u] = True
        cnt += 1
        # ê°„ì„ ì™„í™”
        for w, cost in adj[u]: # ë„ì°©ì •ì  w, ê°€ì¤‘ì¹˜ cost
                    # dist[w] = min(dist[w], dist[u] + cost)
            if dist[w] > dist[u] + cost:
                dist[w] = dist[u] + cost
    return dist[end]
